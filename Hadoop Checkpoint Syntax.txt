## Pig syntax Checkpoint

1. hdfs dfs -put /home/Pig/Pig_Data/artist.json dfs://localhost:9000/pig_data/
   hdfs dfs -put /home/Pig/Pig_Data/movies.json dfs://localhost:9000/pig_data/

2. artist = LOAD 'hdfs://localhost:9000/pig_data/artist.json' using JsonLoader as (id : int, artist_name : chararray, role : chararray, movieid  : int, 										  description : chararray);
   movies = LOAD 'hdfs://localhost:9000/pig_data/movies.json' using JsonLoader as (id : int, movie_name : chararray, year : Date, category:chararray, 										                  director : chararray );

3. movies_by_year = ORDER movies BY year;
   movies_by_year_name = FOREACH movies_by_year generate name, year;

4. american_movies = FILTER movies by category = 'America';

   director_list = FOREACH american_movies generate director;

5. triplets = FOREACH movies generate movieid, id, role;

6. artist_description FOREACH artist generate artist_name, description;

7. movie_group = GROUP BY movieid;
